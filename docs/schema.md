Database Schema Documentation: Administrative System
Este documento detalla la estructura técnica de la base de datos para facilitar el desarrollo de consultas y lógica de negocio.

1. Contabilidad (Core Financials)
1.1 Table: ACCOUNT
Define el plan de cuentas contable y los segmentos financieros vinculados.

-- ACCOUNT definition

CREATE TABLE ACCOUNT (
	ACCOUNTID VARCHAR(16) NOT NULL,
	FINANCIALDESCRIPTION VARCHAR(40),
	FINANCIALID VARCHAR(6) NOT NULL,
	DEPARTMENTID VARCHAR(2),
	DIVISIONID VARCHAR(2),
	FUNDID VARCHAR(2),
	JOBID VARCHAR(4),
	COSTID VARCHAR(4),
	INSERTID INTEGER,
	QBLISTID VARCHAR(25),
	QBNAME VARCHAR(45),
	CONSTRAINT ACCOUNTID11280PK PRIMARY KEY (ACCOUNTID)
);


-- ACCOUNT foreign keys

ALTER TABLE ACCOUNT ADD CONSTRAINT COSTID11280FK FOREIGN KEY (COSTID) REFERENCES COST(COSTID);
ALTER TABLE ACCOUNT ADD CONSTRAINT DEPARTMENTID11280FK FOREIGN KEY (DEPARTMENTID) REFERENCES DEPARTMENT(DEPARTMENTID);
ALTER TABLE ACCOUNT ADD CONSTRAINT DIVISIONID11280FK FOREIGN KEY (DIVISIONID) REFERENCES DIVISION(DIVISIONID);
ALTER TABLE ACCOUNT ADD CONSTRAINT FINANCIALID11280FK FOREIGN KEY (FINANCIALID) REFERENCES FINANCIAL(FINANCIALID);
ALTER TABLE ACCOUNT ADD CONSTRAINT FUNDID11280FK FOREIGN KEY (FUNDID) REFERENCES FUND(FUNDID);
ALTER TABLE ACCOUNT ADD CONSTRAINT JOBID11280FK FOREIGN KEY (JOBID) REFERENCES JOB(JOBID);
